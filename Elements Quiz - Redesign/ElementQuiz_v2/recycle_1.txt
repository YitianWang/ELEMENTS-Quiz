 	 	var slideElement = document.createElement("li"); 
		slideElement.id = "slide" + i;

		var newSlide = document.getElementById("all-slides");
		newSlide.appendChild(slideElement);

		// cover page 
		if (i == 1){

			var titleElement = document.createElement("h1");
 	 		titleElement.appendChild(document.createTextNode(json[i - 1].H1));

 	 		slideElement.appendChild(titleElement);

 	 		var subTitleElement = document.createElement("p");
 	 		subTitleElement.appendChild(document.createTextNode(json[i - 1].H2));
 	 		slideElement.appendChild(subTitleElement);

 	 		var intro = document.createElement("p");
 	 		intro.appendChild(document.createTextNode(json[i - 1].H3));
 	 		slideElement.appendChild(intro);
 	 			 
 	 	}

 	 	// second page 
 	 	if (i == 2) {

 	 		var intro = document.createElement("p");
 	 		intro.appendChild(document.createTextNode(json[i - 1].p));
 	 		slideElement.appendChild(intro);
 	 	}


		if (i != 1 && i != 2) {

			// Question 
			var questionElement = document.createElement("h3");
			questionElement.appendChild(document.createTextNode(json[i - 1].question));
			slideElement.appendChild(questionElement);

			// Form 
			var formElement = document.createElement("form"); 
			slideElement.appendChild(formElement); 
			formElement.id = "form" + i;


			// add Answers to the slide 
			var count = json[i - 1].answers.length;
			for (var j = 1; j <= count; j++) {

				function addRadioButton(text){

					var label = document.createElement("label");
					var radio = document.createElement("input");

					radio.setAttribute("type", "radio");
					radio.setAttribute("name", "radioButton");
					radio.setAttribute("value", j);

					label.appendChild(radio);
					label.appendChild(document.createTextNode(text));

					formElement.appendChild(label);
				}

				addRadioButton(json[i - 1].answers[j - 1]);
	        }

	        // add img the slide 
	        var imgElement = document.createElement("img");
	        imgElement.setAttribute("scr", json[i - 1].img);
	        slideElement.appendChild(imgElement);

	        // add disclaimer
	        var disclaimerElement = document.createElement("p");
	        disclaimerElement.appendChild(document.createTextNode(json[i - 1].disclaimer));
	        disclaimerElement.className = "disclaimer";
	        slideElement.appendChild(disclaimerElement);
		}

			// create new PREVIOUS button 
			var previouButtonElement = document.createElement("button");
			previouButtonElement.appendChild(document.createTextNode("PREVIOUS"));
			previouButtonElement.className = "previous";
			slideElement.appendChild(previouButtonElement);
		}

		}

		var nextButtonElement = document.createElement("button");
		if (i == 1) {	nextButtonElement.appendChild(document.createTextNode("Take the questionnaire >>"));
		}else{	nextButtonElement.appendChild(document.createTextNode("NEXT"));}
		nextButtonElement.className = "next";
		slideElement.appendChild(nextButtonElement);
